# Git Commands to Push Changes

# 1. Check status
git status

# 2. Add all changes
git add .

# 3. Commit with message
git commit -m "v3.0.0 - Full Integration Release"

# 4. Push to GitHub
git push origin main

# 5. Create and push tag
git tag -a v3.0.0 -m "Version 3.0.0 - Full Integration Release"
git push origin v3.0.0
